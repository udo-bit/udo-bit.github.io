        //åœ¨æ–¹æ³•ä¸­ç”¨ StringBuilder æŒ‰ç…§è¦æ±‚è¿›è¡Œæ‹¼æ¥ï¼Œå¹¶æŠŠç»“æœè½¬æˆ String è¿”å›
        StringBuilder sb = new StringBuilder();

        sb.append("[");

        for(int i=0; i<arr.length; i++) {
            if(i == arr.length-1) {
                sb.append(arr[i]);
            } else {
                sb.append(arr[i]).append(", ");
            }
        }

        sb.append("]");

        String s = sb.toString();

        return  s;
    }
