        return id;

        this.id = id;

        return name;

        this.name = name;

        return age;

        this.age = age;
    }

        return "Student{" +
                "id='" + id + '\'' +
                ", name='" + name + '\'' +
                ", age=" + age +
                '}';
    }
}
